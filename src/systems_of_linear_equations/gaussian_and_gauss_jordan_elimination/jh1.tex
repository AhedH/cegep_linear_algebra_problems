\begin{Exercise}[
name={},

difficulty=0,
origin={by Jim Hefferon \cite{JH}}]
Use Gauss's Method to find the unique solution for each system.
\Question
\[
\begin{array}{ccccccccccccccc}
      2x  &+  &3y  &=  &13  \\
          x   &-  &y   &=  &-1
\end{array}
\]

\Question
\[
\begin{array}{ccccccccccccccc}
          x   &  &  &-  &z  &=  &0  \\
          3x  &+ &y &   &   &=  &1  \\
          -x  &+ &y &+  &z  &=  &4
\end{array}
\]

\end{Exercise}

\begin{Answer}
\Question $x=2$, $y=3$
\Question $x=-1$, $y=4$, and $z=-1$.
\end{Answer}
